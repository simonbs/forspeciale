\makeatletter
\define@key{signature}{width}{\def\sig@width{#1}}
\define@key{signature}{content height}{\def\sig@contentHeight{#1}}
\define@key{signature}{image}{\def\sig@image{#1}}
\setkeys{signature}{width=5cm, content height=1cm}%
\newcommand{\signature}[3][]{%
  \begingroup%
  \setkeys{signature}{#1}% Set new keys
  \begin{tabular}{@{}p{\sig@width}@{}}
    #2
    \@ifundefined{sig@image}{
      \\\vspace{\sig@contentHeight}
    }{
    \\
    \parbox[t][\sig@contentHeight][b]{\sig@width}{
      \includegraphics[max height=\sig@contentHeight, max width=\sig@width]{\sig@image}
    }
    }
    \hrule \\[0pt]
    {\small #3} \\[2\normalbaselineskip]
  \end{tabular}
  \endgroup%
}
\makeatother
